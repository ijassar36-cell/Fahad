let index=0,score=0,answered=false;

function loadQuestion(){
  let q=questions[index];
  document.getElementById("question").innerHTML=(index+1)+") "+q.q;
  let ops="";
  ["أ","ب","ج","د"].forEach((l,i)=>{
    ops+=`<div class='option' onclick="choose('${l}',this)"><b>${l})</b> ${q.o[i]}</div>`;
  });
  document.getElementById("options").innerHTML=ops;
  document.getElementById("next").disabled=true;
  answered=false;
}

function choose(letter,el){
  if(answered) return;
  answered=true;

  let correct=questions[index].c;
  let explain=questions[index].e;

  let all=document.querySelectorAll(".option");
  all.forEach(op=>{
    op.style.pointerEvents="none";
    if(op.innerText.trim().startsWith(correct)) op.classList.add("correct");
    else if(op===el) op.classList.add("wrong");
  });

  if(letter===correct) score++;

  document.getElementById("explain-text").innerHTML=explain;
  document.getElementById("popup").style.display="flex";
  document.getElementById("next").disabled=false;
}

function closePopup(){document.getElementById("popup").style.display="none";}

function nextQuestion(){
  if(index<questions.length-1){index++;loadQuestion();}
  else showResult();
}

function showResult(){
  document.getElementById("quiz").style.display="none";
  let r=document.getElementById("result-box");
  r.style.display="block";
  r.innerHTML="نتيجتك: "+score+" من "+questions.length;
}

loadQuestion();
